Create Node.js native test file (`node:test`) that validates MODES array labels and placeholders for jargon-free language:

**Test suite structure:**
1. **No jargon in labels test** — Scans all mode labels for banned terms: "Tech", "Biz", "API", "Deploy" (technical jargon or PM-specific terms)
2. **Verb-led or transformation-clear test** — Validates labels either start with action verbs or show clear transformation (using arrow →)
3. **Placeholder alignment test** — Confirms placeholder text doesn't contain PM language: "dev team", "stakeholder", "leadership"
4. **Placeholder-prompt consistency test** — Validates placeholder audience matches prompt audience (vibe-coder, not PM)

**Challenge:** MODES array is defined inside App.jsx as a const, not exported. Test needs to:
- Read src/App.jsx as text
- Parse MODES array using regex or manual extraction
- Validate extracted labels and placeholders

**Test file template:**
```javascript
const { describe, it } = require('node:test');
const assert = require('node:assert');
const fs = require('fs');
const path = require('path');

describe('UI Label Validation', () => {
  let modes;

  // Parse MODES array from App.jsx
  it('should parse MODES from App.jsx', () => {
    const appContent = fs.readFileSync(path.join(__dirname, '../src/App.jsx'), 'utf-8');
    const modesMatch = appContent.match(/const MODES = \[([\s\S]*?)\];/);
    assert.ok(modesMatch, 'MODES array not found in App.jsx');

    // Extract mode objects (simple regex, may need refinement)
    const modesText = modesMatch[1];
    modes = [];
    const modeRegex = /\{\s*id:\s*'([^']+)',\s*label:\s*'([^']+)',.*?placeholder:\s*"([^"]+)"/g;
    let match;
    while ((match = modeRegex.exec(modesText)) !== null) {
      modes.push({ id: match[1], label: match[2], placeholder: match[3] });
    }
    assert.ok(modes.length > 0, 'No modes parsed');
  });

  it('should not contain jargon in labels', () => {
    const JARGON_TERMS = ['tech', 'biz', 'api', 'deploy'];
    for (const mode of modes) {
      const lower = mode.label.toLowerCase();
      for (const term of JARGON_TERMS) {
        assert.ok(!lower.includes(term), `${mode.id} label contains jargon: "${term}"`);
      }
    }
  });

  it('should not contain PM language in placeholders', () => {
    const PM_TERMS = ['dev team', 'stakeholder', 'leadership'];
    for (const mode of modes) {
      const lower = mode.placeholder.toLowerCase();
      for (const term of PM_TERMS) {
        assert.ok(!lower.includes(term), `${mode.id} placeholder contains PM term: "${term}"`);
      }
    }
  });

  // Additional tests for verb-led format, transformation clarity
});
```

File should be runnable with: `node --test tests/ui-labels.test.js`
